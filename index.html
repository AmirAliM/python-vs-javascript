Control Flow

# Control flow: If else
# https://docs.python.org/3/tutorial/controlflow.html#if-statements

some_number = 3

# Number is 3
if some_number == 1:
    print("Number is 1")
elif some_number == 2:
    print("Number is 2")
elif some_number == 3:
    print("Number is 3")
else:
    print("?")

// Control flow: If else
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/if...else#Using_else_if

const someNumber = 3

// Number is 3
if (someNumber === 1) {
  console.log('Number is 1')
} else if (someNumber === 2) {
  console.log('Number is 2')
} else if (someNumber === 3) {
  console.log('Number is 3')
} else {
  console.log('?')
}
# Control flow: Ternary
x = 2
y = 3

# expr1 if... else expr2
# if True  => return expr1
# if False => return expr2

# even
print("even" if x % 2 == 0 else "odd")
# odd
print("even" if y % 2 == 0 else "odd")

// Control flow: Ternary
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Operators/Conditional_Operator
const x = 2
const y = 3

// condition ? expr1 : expr2
// => expr1 is True, expr2 is False
// if True => return expr1
// if False => return expr2

// even
console.log(x % 2 === 0 ? 'even' : 'odd')
// odd
console.log(y % 2 === 0 ? 'even' : 'odd')

# Control flow: Continue / Break
# https://docs.python.org/3/tutorial/controlflow.html#break-and-continue-statements-and-else-clauses-on-loops

# 1
# 2
# Fizz
# 4
# Buzz
for number in range(1, 100):
    if number == 3:
        print("Fizz")
        continue
    if number == 5:
        print("Buzz")
        break
    print(number)

// Control flow: Continue / Break
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/continue
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/continue

// 1
// 2
// Fizz
// 4
// Buzz

for (let i = 1; i <= 100; i = i + 1) {
  if (i === 3) {
    console.log('Fizz')
    continue
  }

  if (i === 5) {
    console.log('Buzz')
    break
  }

  console.log(i)
}

# String search
# https://docs.python.org/3/library/re.html#re.search
# https://github.com/zeeshanu/learn-regex
import re

# Has a number
# d = digit
if re.search(r"\d", "iphone 8"):
    print("Has a number")

# Doesn't have a number
if not re.search(r"\d", "iphone x"):
    print("Doesn't have a number")

// String search
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Guide/Regular_Expressions
// https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String/match
// https://github.com/zeeshanu/learn-regex

// Has a number
if ('iphone 8'.match(/\d/g)) {
  console.log('Has a number')
}

// Doesn't have a number
// match() returns null
if (!'iphone x'.match(/\d/g)) {
  console.log("Doesn't have a number")
}
